from js import document

def calculate(evt):
    math = float(document.getElementById("math").value)
    english = float(document.getElementById("english").value)
    physics = float(document.getElementById("physics").value)
    
    total = math + english + physics
    failed_subjects = []
    for sub, mark in zip(["Math","English","Physics"], [math, english, physics]):
        if mark < 32:
            failed_subjects.append(sub)
    
    output = ""
    if failed_subjects:
        output += f"Result: Failed\nFailed in: {', '.join(failed_subjects)}"
    else:
        percentage = total/3
        if percentage >= 90: grade="A+"
        elif percentage >= 80: grade="A"
        elif percentage >= 70: grade="B"
        elif percentage >= 60: grade="C"
        elif percentage >= 50: grade="D"
        else: grade="E"
        output += f"Result: Passed\nPercentage: {percentage}\nGrade: {grade}"
    
    document.getElementById("output").innerText = output

document.getElementById("calculate").addEventListener("click", calculate)
