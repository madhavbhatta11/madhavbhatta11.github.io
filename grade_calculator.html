from js import document

def calculate(event):

    math = float(document.getElementById("math").value)
    english = float(document.getElementById("english").value)
    physics = float(document.getElementById("physics").value)

    total = math + english + physics

    failed_subjects = []

    if math < 32:
        failed_subjects.append("Math")

    if english < 32:
        failed_subjects.append("English")

    if physics < 32:
        failed_subjects.append("Physics")

    output = ""

    if failed_subjects:
        output = "Result: Failed\n"
        output += "Failed in: " + ", ".join(failed_subjects)

    else:

        percentage = total / 3

        if percentage >= 90:
            grade = "A+"
        elif percentage >= 80:
            grade = "A"
        elif percentage >= 70:
            grade = "B"
        elif percentage >= 60:
            grade = "C"
        elif percentage >= 50:
            grade = "D"
        else:
            grade = "E"

        output = f"Result: Passed\nPercentage: {percentage:.2f}\nGrade: {grade}"

    document.getElementById("output").innerText = output


document.getElementById("calculate").addEventListener("click", calculate)
